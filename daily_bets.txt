function parseBets(raw) {
  const lines = raw.split('\n');

  const bets = [];

  for (const line of lines) {
    if (line.includes('cote')) {
      // ex: "• Equipe B (1N2) – cote 1.85"
      const clean = line.replace(/•/g, '').trim();

      // Séparer : description vs cote
      const parts = clean.split('cote');

      const description = parts[0].trim().replace(/[-–]/g, '').trim();
      const odds = parts[1]?.trim();

      bets.push({ description, odds });
    }
  }

  return bets;
}
